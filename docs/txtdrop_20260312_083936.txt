### 2단계: API 호출 방식 선택하기

#### 🔵 **방법 A: HTTP 직접 호출 (가장 간단!)**

러브타입 앱에서 바로 사용할 수 있는 코드예요:

```javascript
// JavaScript 예제 (웹앱용)
async function getDeepseekResponse(userPrompt) {
  const API_KEY = "여기에_발급받은_API키_넣기";
  const url = "https://api.deepseek.com/v1/chat/completions";  // 엔드포인트 주소 
  
  const headers = {
    "Content-Type": "application/json",
    "Authorization": `Bearer ${API_KEY}`  // Bearer 다음에 꼭 띄어쓰기! 
  };
  
  const data = {
    model: "deepseek-chat",  // V3 모델 (빠름, 저렴) 
    // model: "deepseek-reasoner",  // R1 모델 (추론 강함, 약간 느림) 
    messages: [
      { role: "system", content: "당신은 연애 심리 분석 전문가입니다. 재미있고 공감가는 톤으로 답변해주세요." },
      { role: "user", content: userPrompt }
    ],
    temperature: 0.8,  // 창의성 조절 (0.7~0.9 추천) 
    max_tokens: 300,   // 응답 길이 제한 
    stream: false      // false면 한번에 받음, true면 스트리밍 
  };
  
  try {
    const response = await fetch(url, {
      method: "POST",
      headers: headers,
      body: JSON.stringify(data)
    });
    
    if (!response.ok) {
      throw new Error(`HTTP error! status: ${response.status}`);
    }
    
    const result = await response.json();
    return result.choices[0].message.content;  // 여기서 답변 추출! 
  } catch (error) {
    console.error("API 호출 실패:", error);
    return "죄송해요, 지금은 연애 상담사가 바빠요. 잠시 후 다시 시도해주세요! 😅";
  }
}

// 사용 예시
const 해설 = await getDeepseekResponse("ENFP 성향인 사람의 연애 스타일을 분석해줘");
```

#### 🔵 **방법 B: OpenAI SDK 사용 (더 편리!)**

딥시크는 OpenAI API와 **완전 호환**돼서 OpenAI SDK로 바로 쓸 수 있어요 :

```javascript
// OpenAI SDK 설치 필요: npm install openai
import OpenAI from 'openai';

const deepseek = new OpenAI({
  apiKey: "여기에_API키",
  baseURL: "https://api.deepseek.com/v1",  // 이거 중요! 
  dangerouslyAllowBrowser: true  // 웹앱에서 직접 호출할 때 필요
});

async function getLoveTypeAnalysis(type, scores) {
  const response = await deepseek.chat.completions.create({
    model: "deepseek-chat",
    messages: [
      { role: "system", content: "당신은 연애 심리 전문가입니다. MBTI 유형별 연애 스타일을 재미있게 분석해주세요." },
      { role: "user", content: `${type} 유형이야. E점수 ${scores.E}점, I점수 ${scores.I}점이야. 이 사람의 연애 스타일을 분석해줘.` }
    ],
    temperature: 0.8,
    max_tokens: 250
  });
  
  return response.choices[0].message.content;
}
```

### 4단계: 캐싱 전략 적용 (오빠 아이디어 완벽 구현!)

```javascript
// 간단한 캐시 시스템
const API_CACHE = {
  dailyCount: 0,
  maxDaily: 30,  // 하루 30회 제한
  lastReset: new Date().toDateString(),
  cacheStore: {}  // { "ENFP": ["해설1", "해설2"], ... }
};

function getFromCache(type) {
  // 날짜 바뀌면 카운트 리셋
  const today = new Date().toDateString();
  if (API_CACHE.lastReset !== today) {
    API_CACHE.dailyCount = 0;
    API_CACHE.lastReset = today;
  }
  
  // 캐시에 있으면 랜덤으로 하나 반환
  if (API_CACHE.cacheStore[type] && API_CACHE.cacheStore[type].length > 0) {
    const cacheList = API_CACHE.cacheStore[type];
    return cacheList[Math.floor(Math.random() * cacheList.length)];
  }
  
  return null;
}

async function getLoveTypeDescription(type, scores) {
  // 1. 캐시 확인
  const cached = getFromCache(type);
  if (cached) return cached;
  
  // 2. 일일 제한 확인
  if (API_CACHE.dailyCount >= 30) {
    // 제한 초과시 기본 해설 반환
    return getBasicDescription(type, scores);
  }
  
  // 3. API 호출
  try {
    API_CACHE.dailyCount++;
    const result = await callDeepseekAPI(type, scores);
    
    // 4. 캐시에 저장
    if (!API_CACHE.cacheStore[type]) {
      API_CACHE.cacheStore[type] = [];
    }
    API_CACHE.cacheStore[type].push(result);
    
    return result;
  } catch (error) {
    console.error("API 오류:", error);
    return getBasicDescription(type, scores);
  }
}
```
